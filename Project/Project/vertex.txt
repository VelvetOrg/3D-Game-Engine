attribute vec3 vertPosition;
attribute vec2 textureCoords;

varying vec2 coords;
varying vec3 colour;

uniform mat4 viewProjection;
uniform vec3 modelColour;
uniform mat4 model;

void main() 
{
	gl_Position = viewProjection * model * vec4(vertPosition, 1.0);

	coords = textureCoords;
	colour = modelColour;
}
